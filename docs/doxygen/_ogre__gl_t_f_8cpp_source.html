108  FileType detectType(const std::string& path) const
109  {
110  //Quickly open the file as binary and chekc if there's the gltf binary magic number
111  {
112  auto probe = std::ifstream(path, std::ios_base::binary);
113  if(!probe) throw std::runtime_error("Could not open " + path);
114 
115  std::array<char, 5> buffer{};
116  for(size_t i{ 0 }; i < 4; ++i) probe >> buffer[i];
117  buffer[4] = 0;
118 
119  if(std::string("glTF") == std::string(buffer.data())) {
120  //OgreLog("Detected binary file thanks to the magic number at the start!");
121  return FileType::Binary;
122  }
123  }
124 
125  //If we don't have any better, check the file extension.
126  auto extension = path.substr(path.find_last_of('.') + 1);
127  std::transform(std::begin(extension), std::end(extension), std::begin(extension), [](char c) { return char(tolower(int(c))); });
128  if(extension == "gltf") return FileType::Ascii;
129  if(extension == "glb") return FileType::Binary;
130 
131  return FileType::Unknown;
132  }

135  bool loadInto(loaderAdapter& adapter, const std::string& path)
136  {

144  case FileType::Binary:
